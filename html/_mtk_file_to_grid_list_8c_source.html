174 
175  /* Get HDF file identifier. */
176  hdf_status_code = EHidinfo(fid, &HDFfid, &sid);
177  if (hdf_status_code == FAIL)
178  return FAIL;
179 
180  /* Start Vgroup Interface */
181  /* ---------------------- */
182  Vstart(HDFfid);
183 
184 
185  /* If string buffer size is requested then zero out counter */
186  /* -------------------------------------------------------- */
187  if (strbufsize != NULL)
188  {
189  *strbufsize = 0;
190  }
191  /* Search for objects from begining of HDF file */
192  /* -------------------------------------------- */
193  vgRef = -1;
194 
195  /* Loop through all objects */
196  /* ------------------------ */
197  while (1)
198  {
199  /* Get Vgroup reference number */
200  /* --------------------------- */
201  vgRef = Vgetid(HDFfid, vgRef);
202 
203  /* If no more then exist search loop */
204  /* --------------------------------- */
205  if (vgRef == -1)
206  {
207  break;
208  }
209  /* Get Vgroup ID, name, and class */
210  /* ------------------------------ */
211  vGrpID = Vattach(HDFfid, vgRef, "r");
212  Vgetname(vGrpID, name);
213  Vgetclass(vGrpID, class);
214 
215 
216  /* If object of desired type (GRID) ... */
217  /* -------------------------------------------------- */
218  if (strcmp(class, "GRID") == 0)
219  {
220 
221  /* Increment counter */
222  /* ----------------- */
223  nobj++;
224 
225 
226  /* If object list requested add name to list */
227  /* ----------------------------------------- */
228  if (objectlist != NULL)
229  {
230  if (nobj == 1)
231  {
232  strcpy(objectlist, name);
233  } else
234  {
235  strcat(objectlist, ",");
236  strcat(objectlist, name);
237  }
238  }
239  /* Compute string length of object entry */
240  /* ------------------------------------- */
241  slen = (nobj == 1) ? strlen(name) : strlen(name) + 1;
242 
243 
244  /* If string buffer size is requested then increment buffer size */
245  /* ------------------------------------------------------------- */
246  if (strbufsize != NULL)
247  {
248  *strbufsize += slen;
249  }
250  }
251  /* Detach Vgroup */
252  /* ------------- */
253  Vdetach(vGrpID);
254  }
255 
256  /* "Close" Vgroup interface and HDFEOS file */
257  /* ---------------------------------------- */
258  Vend(HDFfid);
259 
260  return (nobj);
261 }
