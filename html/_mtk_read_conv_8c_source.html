116  int32 start[10]; /* HDF-EOS start dimension */
117  int32 edge[10]; /* HDF-EOS edge dimension */

120  int32 dims[10]; /* HDF-EOS dimensions */

122  char *basefield = NULL; /* Base fieldname */
123  int nextradims; /* Number of extra dimensions */
124  int *extradims = NULL; /* Extra dimension list */
125  int i; /* Loop index */
126  int32 xdimsize; /* X dimension size of file */
127  int32 ydimsize; /* Y dimension size of file */
128  float64 ulc[2]; /* Upper left corner in meters */
129  float64 lrc[2]; /* Lower right corner in meters */
130  char *dataptr; /* Data pointer */
131  int32 HDFfid; /* HDF file identifier (not used) */
132  int32 sid; /* HDF SD identifier */

134  if (gridname == NULL) return MTK_NULLPTR;
135  if (fieldname == NULL) return MTK_NULLPTR;
136 
137  /* Get the HDF SD identifier. */
138  hdfstatus = EHidinfo(fid, &HDFfid, &sid);
139  if (hdfstatus == FAIL)
140  MTK_ERR_CODE_JUMP(MTK_HDFEOS_EHIDINFO_FAILED);

142  /* Determine path of this file */
143  status = MtkFileToPathFid(sid, &path);
144  MTK_ERR_COND_JUMP(status);

147  status = MtkFileGridToResolutionFid(fid, gridname, &resolution);
148  MTK_ERR_COND_JUMP(status);
149

151  status = MtkSnapToGrid(path, resolution, region, &map);
152  MTK_ERR_COND_JUMP(status);
153 
154  /* Attach to grid */
155  gid = GDattach(fid, (char*)gridname);
156  if (gid == FAIL) MTK_ERR_CODE_JUMP(MTK_HDFEOS_GDATTACH_FAILED);
157

159  hdfstatus = GDgridinfo(gid, &xdimsize, &ydimsize, ulc, lrc);
160  if (hdfstatus == FAIL) MTK_ERR_CODE_JUMP(MTK_HDFEOS_GDGRIDINFO_FAILED);
161 
162  /* Parse fieldname for extra dimensions */
163  status = MtkParseFieldname(fieldname, &basefield, &nextradims, &extradims);
164  MTK_ERR_COND_JUMP(status);
165 
166  /* Determine rank, dimensions and datatype of the field */
167  hdfstatus = GDfieldinfo(gid, basefield, &rank, dims, &hdf_datatype, dimlist);
168  if (hdfstatus == FAIL) MTK_ERR_CODE_JUMP(MTK_HDFEOS_GDFIELDINFO_FAILED);
169

171  if (rank != nextradims + 2) MTK_ERR_CODE_JUMP(MTK_BAD_ARGUMENT);
172 
173  /* Convert hdf datatype to mtk datatype */
174  status = MtkHdfToMtkDataTypeConvert(hdf_datatype, &datatype);

176 
177  /* Create a map structure for the contents of the file, then we can */
178  /* determine the coordinates of the intersecting map area we need to load */
179  /* Note - ulc y and lrc y are flipped. */
180  mapfile = map;
181  mapfile.nline = dims[0];
182  mapfile.nsample = dims[1];
183  mapfile.som.ulc.x = ulc[0];
184  mapfile.som.ulc.y = lrc[1];
185  mapfile.som.lrc.x = lrc[0];
186  mapfile.som.lrc.y = ulc[1];
187  mapfile.som.ctr.x = mapfile.som.ulc.x +
188  ((mapfile.som.lrc.x - mapfile.som.ulc.x) / 2.0);
189  mapfile.som.ctr.y = mapfile.som.ulc.y +
190  ((mapfile.som.lrc.y - mapfile.som.ulc.y) / 2.0);
191

193  /* If out of bounds then clip to size of file coordinates. */
194  /* The status of this routine can be ignored because we catch */
195  /* out of bounds when line or sample is -1.0. */
196  /* Also, subtract half a pixel to center coordinate. */

209  status = MtkLSToSomXY(mapfile, (float)(ulc_line + 0.5), (float)(ulc_sample + 0.5), &ulc_som.x, &ulc_som.y);

214  /* The status of this routine can be ignored because we catch */
215  /* out of bounds when line or sample is -1.0. */
216  /* Also, subtract half a pixel to center coordinate. */

218 
219  line -= 0.5;
220  sample -= 0.5;
221 
222  if (line < 0.0) lrc_line = dims[0] - 1;
223  else lrc_line = (int)line;
224 
225  if (sample < 0.0) lrc_sample = dims[1] - 1;
226  else lrc_sample = (int)sample;
227 
228  /* Determine som x/y coordinates of lrc intersecting coordinate */
229  status = MtkLSToSomXY(mapfile, (float)(lrc_line + 0.5), (float)(lrc_sample + 0.5), &lrc_som.x, &lrc_som.y);
230  MTK_ERR_COND_JUMP(status);

235 
236  start[0] = ulc_line;
237  start[1] = ulc_sample;
238 
239  edge[0] = nline;
240  edge[1] = nsample;
241 
242  for (i = 2; i < rank; i++) {
243  start[i] = extradims[i-2];
244  edge[i] = 1;
245  }
246 
247  /* Allocate the temp data buffer */
248  status = MtkDataBufferAllocate(nline, nsample, datatype, &filebuf_tmp);

252  hdfstatus = GDreadfield(gid, basefield, start, NULL, edge,

262  MTK_ERR_COND_JUMP(status);
263

268  for (l = 0; l < map.nline; l++) {
269  for (s = 0; s < map.nsample; s++) {
270  memcpy((void *)dataptr, fill.dataptr, buf.datasize);
271  dataptr += buf.datasize;
272  }
273  }
274  MtkDataBufferFree(&fill);
275  }
276

278  i = 0;
279  for (x = ulc_som.x; x <= lrc_som.x; x += map.resolution) {
280  MtkSomXYToLS(map, x, ulc_som.y, &line, &sample);
281  memcpy((void *)&(buf.data.u8[(int)line][(int)sample]),
282  (void *)&(filebuf_tmp.data.u8[i++][0]),
283  nsample * buf.datasize);
284  }
285 
286  free(basefield);
287  free(extradims);
288  MtkDataBufferFree(&filebuf_tmp);
289 
290  *databuf = buf;
291  *mapinfo = map;
292 
293  return MTK_SUCCESS;
294  ERROR_HANDLE:
295  if (gid != FAIL) GDdetach(gid);
296  if (basefield != NULL) free(basefield);
297  if (extradims != NULL) free(extradims);
298  MtkDataBufferFree(&filebuf_tmp);
299  MtkDataBufferFree(&buf);
300  return status_code;
301 }
